<!--  Tab类 -->
<script>
;(function($) {
    var Tab = function(tab) {
        var _this = this
        // 保存单个tab组件
        this.tab = tab
        // 配置默认参数
        this.config = {
            triggerType: 'mouseover',  // 鼠标触发类型click,和mouseover
            effect: 'default', // 内容切换效果，切换动画类型
            invoke: 1, // 当前默认显示的tab
            auto: false // tab是否自动切换，如果有时间，则是自动
        }
        // 获得最终的配置参数
        if(this.getConfig()) {
            $.extend(this.config, this.getConfig())
        }
        // 保存标签和内容列表(dom 列表)
        this.tabItems = this.tab.find('nav.tab-nav a')
        this.contentItems = this.tab.find('div.tab-content div.content-item')

        // 保存配置参数
        var config = this.config

        // 给dom绑定事件
        if(config.triggerType === 'click') {
            this.tabItems.bind(config.triggerType, function() {
                _this.invoke($(this))
            })
        } else {
            this.tabItems.find('mouseover', function() {
                _this.invoke($(this))
            })
        }
        // 自动切换功能: 每隔一个时间间隔，去触发一下绑定的事件
        if(config.auto) {
            this.timer = null;
            // 计数器
            this.loop = 0;
            this.autoPlay()
            // tab移入定时器清除，移出开始
            this.tab.hover(function() {
                clearInterval(_this.timer)
            }, function() {
                this.autoPlay()
            })
        }

        // 设置默认显示第几个
        if(config.invoke > 1) {
            this.invoke(this.tabItems.eq(config.invoke - 1))
        }
    }

    Tab.prototype = {
        // 获取配置参数
        getConfig: function() {
            // 拿tab elem节点上的data-config
            var config = this.tab.attr('data-config')
            if(config && config !== '') {
                return $.parseJSON(config)
            } else {
                return null
            }
        },
        // 事件驱动函数
        invoke: function(curTab) {
            // 标签的选中状态, 当前选中的标签加上actived样式，其兄弟元素去掉
            // 切换对应内容，根据配置参数effect确定是怎么切换的
            var _this = this
            var index = curTab.index()
            var config = this.config
            curTab.addClass('actived').siblings().removeClass('actived')
            // 切换内容区域
            if(config.effect === 'fade') {
                this.contentItems.eq(index).fadeIn().siblings().fadeOut()
            } else {
                // default
                this.contentItems.eq(index).addClass('current').siblings().removeClass('current')
            }
            // 自动切换时，开始轮播的index要更改
            if(config.auto) {
                this.loop = index
            }
        },
        // 自动间隔时间切换
        autoPlay: function() {
            var _this = this
            var config = this.config
            var tabItems = this.tabItems
            var tabLength = tabItems.size()

            this.timer = setInterval(function() {
                _this.loop++
                if(_this.loop >= tabLength) {
                    _this.loop = 0
                }
                tabItems.eq(_this.loop).trigger(config.trggerType)
            }, config.auto)
        }
    }

    // 初始化所有tab
    Tab.init = function(tabs) {
        // _this 是 Tab这个类
        var _this = this
        tabs.each(function() {
            // 这个this是each里面的每一个tab
            new _this($(this))
        })
    }

    // 注册成jq方法, 就是在fn上去扩展一个方法
    $.fn.extent({
        tab: function() {
            // this是$('.jq-tab').tab()实例化的时候，$('.jq-tab')获取到的集合
            this.each(function() {
                // $(this)中的this是集合中的每一个
                new Tab($(this))
            })
            // 返回this，可以链式调用
            return this
        }
    })
    window.Tab = tab
})(jq)
</script>

# 移动端web组件 Dialog对话框

DOM: scrollHeight, clientHeight, createElement, appendChild, removeChild 页面的宽高：const sH = document.documentElement.scrollHeight/scrollWidth 可是区域宽高：const cH = document.documentElement.clientHeight/clientWidth(定位内容中心用) 弹出层：创建一个遮罩层 maskDom，宽高为页面的宽高 maskDom.style.height= sH + 'px', document.body.appendChild(maskDom); 点击遮罩层删除自身和弹窗内容 document.body.removeChild(maskDom) 创建内容 mainDom, 将之定位, 设置定位位置，left=(可视区域的宽度-自身宽度)/2 + 'px'，可视区域的宽度 = 页面的宽; top=(可视区域的高度-自身高度)/2 + 'px'

<script>
;(function($) {
    var Dialog = function(config) {
        // 这里的this指的是 new 的实例
        var _this = this;
        // 默认配置参数
        this.config = {
            type: 'warning', // 对话框类型，loading, ok, warning, icon不同
            buttons: null,   // 按钮组
            delay: null, // 弹出框多久自动关闭
            message: null,
            width: 'auto',
            height: 'auto',
            maskOpacity: null
        }
        // 以对象字面量形式传过来的对象
        if(config && $.isPlainObject(config)) {
            $.extend(this.config, config)
        } else {
            this.isConfig = true
        }

        // 创建基本的dom
        this.body = $('body')
        this.mask = $('<div class="g-dialog-contianer">') // 遮罩层
        this.win = $('<div class="dialog-window">') // 弹出框
        this.winHeader = $('<div class="dialog-header">') // 弹出框
        this.winContent = $('<div class="dialog-content">') // 弹出框
        this.winFooter = $('<div class="dialog-footer">') // 弹出框

        // 渲染DOM
        this.creat()
    }

    Dialog.prototype = {
        // 这里的this指的是 Dialog, 这个类 ?
        // 创建DOM
        creat: function() {
            var _this = this
            var config = this.config
            var mask = this.mask
            var win = this.win
            var header = this.winHeader
            var content = this.winContent
            var footer = this.winFooter
            var body = this.body

            // 如果没有传递任何配置参数
            if(this.isConfig) {
                win.append(header.addClass('waiting'))
                mask.append(win)
                body.append(mask)
            } else {
                // 根据配置文件创建响应弹窗
                header.addClass(config.type)
                win.append(header)
                // message
                if(config.message) {
                    win.append(content.html(config.message))
                }
                // buttons
                if(config.buttons) {
                    win.append(footer)
                }
                mask.append(win)
                body.append(mask)
                // 设置宽高
                if(config.width !== 'auto') {
                    win.width(config.width)
                }
                if(config.height !== 'auto') {
                    win.height(config.height)
                }
                // mask透明度
                if(config.maskOpacity) {
                    mask.css('backgroundColor', 'ragb(0,0,0,'+config.maskOpacity+')')
                }
                // 自动关闭
                if(config.delay && config.delay !== 0) {
                    window.setTimeout(function() {
                        // 在这个函数体内访问this就访问不到了,所以缓存为_this
                        _this.close()
                    }, config.delay)
                }
            }
        },

        close: function() {
            this.mask.remove()
        }
    }

    window.Dialog = Dialog;
})(Zepto)
</script>
